413static inline Quat QuatSlerp(const Quat q1, const Quat q2, float percent)
414{
415 //percent range: 0-1
416 //https://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/slerp/index.htm
417 float dot = q1.W*q2.W + q1.X*q2.X + q1.Y*q2.Y + q1.Z*q2.Z;
418 // if qa=qb or qa=-qb then theta = 0 and we can return qa
419 if (abs(dot) >= 1.f)
420 {
421 return q1;
422 }
423 // Calculate temporary values.
424 float halfTheta = acosf(dot);
425 float sinHalfTheta = sqrtf(1.f - dot*dot);
426 // if theta = 180 degrees then result is not fully defined
427 // we could rotate around any axis normal to qa or qb
428 if (fabsf(sinHalfTheta) < 0.001f)
429 {
430 Quat q;
431 q.W = q1.W * 0.5f + q2.W * 0.5f;
432 q.X = q1.X * 0.5f + q2.X * 0.5f;
433 q.Y = q1.Y * 0.5f + q2.Y * 0.5f;
434 q.Z = q1.Z * 0.5f + q2.Z * 0.5f;
435 return q;
436 }
437 float ratioA = sinf((1.f - percent) * halfTheta) / sinHalfTheta;
438 float ratioB = sinf(percent * halfTheta) / sinHalfTheta;
439
440 Quat q;
441 q.W = (q1.W * ratioA + q2.W * ratioB);
442 q.X = (q1.X * ratioA + q2.X * ratioB);
443 q.Y = (q1.Y * ratioA + q2.Y * ratioB);
444 q.Z = (q1.Z * ratioA + q2.Z * ratioB);
445
446 return q;
447}

461static inline Quat RotatorToQuat(const Rotator rot)
462{
463 float rotatorToRadian = ((CONST_PI_F / 180.f) * .5f) / CONST_DegToUnrRot;
464 float sinPitch = sinf(rot.Pitch * rotatorToRadian);
465 float cosPitch = cosf(rot.Pitch * rotatorToRadian);
466 float sinYaw = sinf(rot.Yaw * rotatorToRadian);
467 float cosYaw = cosf(rot.Yaw * rotatorToRadian);
468 float sinRoll = sinf(rot.Roll * rotatorToRadian);
469 float cosRoll = cosf(rot.Roll * rotatorToRadian);
470
471 Quat convertedQuat;
472 convertedQuat.X = (cosRoll * sinPitch * sinYaw) - (sinRoll * cosPitch * cosYaw);
473 convertedQuat.Y = (-cosRoll * sinPitch * cosYaw) - (sinRoll * cosPitch * sinYaw);
474 convertedQuat.Z = (cosRoll * cosPitch * sinYaw) - (sinRoll * sinPitch * cosYaw);
475 convertedQuat.W = (cosRoll * cosPitch * cosYaw) + (sinRoll * sinPitch * sinYaw);
476
477 return convertedQuat;
478}

480static inline Rotator QuatToRotator(const Quat q)
481{
482 //From Arator's HelperFunctions
483 Vector fwd = RotateVectorWithQuat(Vector(1, 0, 0), q);
484 Vector right = RotateVectorWithQuat(Vector(0, 1, 0), q);
485 Vector up = RotateVectorWithQuat(Vector(0, 0, 1), q);
486 fwd.normalize();
487 right.normalize();
488 up.normalize();
489
490 //Pitch
491 float pitch_f = asinf(fwd.Z);
492 int PITCH = static_cast<int>((pitch_f / (CONST_PI_F / 2.f)) * 16384.f);
493
494 //Yaw
495 float hor_mag = sqrtf(fwd.X * fwd.X + fwd.Y * fwd.Y);
496 float fwd_y = asinf(fwd.Y / hor_mag);
497 if (fwd_y >= 0)
498 {
499 if (fwd.X < 0)
500 fwd_y = CONST_PI_F - fwd_y;
501 }
502 else
503 {
504 if (fwd.X < 0)
505 fwd_y = -CONST_PI_F - fwd_y;
506 }
507 int YAW = static_cast<int>((fwd_y / CONST_PI_F) * 32768.f);
508
509 //Roll
510 Vector vert = (up.Z < 0) ? Vector(0, 0, -1) : Vector(0, 0, 1);
511 Vector hor_right = Vector::cross(fwd, vert) * -1;// left-handed coordinate system
512 hor_right.normalize();
513
514 float roll_cos = Vector::dot(hor_right, right);
515 float roll_f = acosf(roll_cos);
516 float up_f = asinf(up.Z);
517
518 if (right.Z >= 0)
519 {
520 if (up.Z >= 0)
521 roll_f = -roll_f;
522 else
523 roll_f = -CONST_PI_F + roll_f;
524 }
525 else
526 {
527 if (up.Z < 0)
528 roll_f = CONST_PI_F - roll_f;
529 }
530 int ROLL = static_cast<int>((roll_f / CONST_PI_F) * 32768.f);
531
532 return Rotator(PITCH, YAW, ROLL);
533}
